1//! # LibYML (a fork of unsafe-libyaml)

10//!
11//! LibYML is a Rust library for working with YAML data, forked from [unsafe-libyaml][01]. It offers a safe and efficient interface for parsing, emitting, and manipulating YAML data.
12//!
13//! ## Features
14//!
15//! - **Serialization and Deserialization**: Easy-to-use APIs for serializing Rust structs and enums to YAML and vice versa.
16//! - **Custom Struct and Enum Support**: Seamless serialization and deserialization of custom data types.
17//! - **Comprehensive Error Handling**: Detailed error messages and recovery mechanisms.
18//! - **Streaming Support**: Efficient processing of large YAML documents.
19//! - **Alias and Anchor Support**: Handling of complex YAML structures with references.
20//! - **Tag Handling**: Support for custom tags and type-specific serialization.
21//! - **Configurable Emitter**: Customizable YAML output generation.
22//! - **Extensive Documentation**: Detailed docs and examples for easy onboarding.
23//! - **Safety and Efficiency**: Minimized unsafe code with an interface designed to prevent common pitfalls.
24//!
25//! ## Installation
26//!
27//! Add this to your `Cargo.toml`:
28//!
29//! ```toml
30//! [dependencies]
31//! libyml = "0.0.5"
32//! ```
33//!
34//! ## Documentation
35//!
36//! For full API documentation, please visit [https://doc.libyml.com/libyml/][03] or [https://docs.rs/libyml][07].

101/// Extern functions and macros for interacting with the underlying libyaml C library.
102///
103/// This module includes low-level functions for memory allocation, comparison, and
104/// movement that bridge Rust and C. It also contains some internal macros for
105/// asserting conditions in a way that integrates with the C code.
106#[macro_use]
107pub mod externs {
108    use crate::libc;
109    use crate::ops::{die, ForceAdd as _, ForceInto as _};
110    use alloc::alloc::{self as rust, Layout};
111    use core::mem::MaybeUninit;
112    use core::mem::{align_of, size_of};
113    use core::ptr;
114    use core::slice;

116    const HEADER: usize = {
117        let need_len = size_of::<usize>();
118        (need_len + MALLOC_ALIGN - 1) & !(MALLOC_ALIGN - 1)
119    };
120
121    const MALLOC_ALIGN: usize = {
122        let int_align = align_of::<libc::c_ulong>();
123        let ptr_align = align_of::<usize>();
124        if int_align >= ptr_align {
125            int_align
126        } else {
127            ptr_align
128        }
129    };
130
131    /// Allocates memory.
132    ///
133    /// # Safety
134    ///
135    /// This function is unsafe because it directly manipulates raw memory. The caller must ensure that
136    /// the allocated memory is properly managed and freed when no longer needed.
137    pub unsafe fn malloc(size: libc::c_ulong) -> *mut libc::c_void {
138        let size = HEADER.force_add(size.force_into());
139        let layout = Layout::from_size_align(size, MALLOC_ALIGN)
140            .ok()
141            .unwrap_or_else(die);
142        let memory = rust::alloc(layout);
143        if memory.is_null() {
144            return die();
145        }
146        memory.cast::<usize>().write(size);
147        memory.add(HEADER).cast()
148    }
149
150    /// Reallocates memory.
151    ///
152    /// # Safety
153    ///
154    /// This function is unsafe because it directly manipulates raw memory.
155    /// The caller must ensure that the original memory was allocated by `malloc`.
156    pub(crate) unsafe fn realloc(
157        ptr: *mut libc::c_void,
158        new_size: libc::c_ulong,
159    ) -> *mut libc::c_void {
160        let mut memory = ptr.cast::<u8>().sub(HEADER);
161        let size = memory.cast::<usize>().read();
162        let layout =
163            Layout::from_size_align_unchecked(size, MALLOC_ALIGN);
164        let new_size = HEADER.force_add(new_size.force_into());
165        memory = rust::realloc(memory, layout, new_size);
166        if memory.is_null() {
167            return die();
168        }
169        memory.cast::<usize>().write(new_size);
170        memory.add(HEADER).cast()
171    }
172
173    /// Frees allocated memory.
174    ///
175    /// # Safety
176    ///
177    /// This function is unsafe because it deallocates memory pointed to by `ptr`.
178    /// The caller must ensure that `ptr` was previously allocated by `malloc` or `realloc`.
179    pub unsafe fn free(ptr: *mut libc::c_void) {
180        let memory = ptr.cast::<u8>().sub(HEADER);
181        let size = memory.cast::<usize>().read();
182        let layout =
183            Layout::from_size_align_unchecked(size, MALLOC_ALIGN);
184        rust::dealloc(memory, layout);
185    }
186
187    /// Compares two memory blocks.
188    ///
189    /// # Safety
190    ///
191    /// This function is unsafe because it compares raw memory. The caller must ensure the pointers
192    /// and size are correct.
193    pub(crate) unsafe fn memcmp(
194        lhs: *const libc::c_void,
195        rhs: *const libc::c_void,
196        count: libc::c_ulong,
197    ) -> libc::c_int {
198        let lhs =
199            slice::from_raw_parts(lhs.cast::<u8>(), count as usize);
200        let rhs =
201            slice::from_raw_parts(rhs.cast::<u8>(), count as usize);
202        lhs.cmp(rhs) as libc::c_int
203    }
204
205    /// Copies memory from `src` to `dest`.
206    ///
207    /// # Safety
208    ///
209    /// The caller must ensure that the memory areas do not overlap and that the destination is large
210    /// enough to hold the data.
211    pub(crate) unsafe fn memcpy(
212        dest: *mut libc::c_void,
213        src: *const libc::c_void,
214        count: libc::c_ulong,
215    ) -> *mut libc::c_void {
216        if dest.is_null() || src.is_null() {
217            return die();
218        }
219        ptr::copy_nonoverlapping(
220            src.cast::<MaybeUninit<u8>>(),
221            dest.cast::<MaybeUninit<u8>>(),
222            count as usize,
223        );
224        dest
225    }
226
227    /// Moves memory from `src` to `dest`.
228    ///
229    /// # Safety
230    ///
231    /// The caller must ensure that the memory areas do not overlap or are correctly managed.
232    pub unsafe fn memmove(
233        dest: *mut libc::c_void,
234        src: *const libc::c_void,
235        count: libc::c_ulong,
236    ) -> *mut libc::c_void {
237        if dest.is_null() || src.is_null() {
238            return die();
239        }
240        ptr::copy(
241            src.cast::<MaybeUninit<u8>>(),
242            dest.cast::<MaybeUninit<u8>>(),
243            count as usize,
244        );
245        dest
246    }

248    /// Fills memory with a constant byte.
249    ///
250    /// # Safety
251    ///
252    /// The caller must ensure the memory is valid and the length is correct.
253    pub unsafe fn memset(
254        dest: *mut libc::c_void,
255        ch: libc::c_int,
256        count: libc::c_ulong,
257    ) -> *mut libc::c_void {
258        ptr::write_bytes(dest.cast::<u8>(), ch as u8, count as usize);
259        dest
260    }

262    /// Compares two strings.
263    ///
264    /// # Safety
265    ///
266    /// The caller must ensure the strings are null-terminated and valid.
267    pub(crate) unsafe fn strcmp(
268        lhs: *const libc::c_char,
269        rhs: *const libc::c_char,
270    ) -> libc::c_int {
271        if lhs.is_null() || rhs.is_null() {
272            return die();
273        }
274        let lhs = slice::from_raw_parts(
275            lhs.cast::<u8>(),
276            strlen(lhs) as usize,
277        );
278        let rhs = slice::from_raw_parts(
279            rhs.cast::<u8>(),
280            strlen(rhs) as usize,
281        );
282        lhs.cmp(rhs) as libc::c_int
283    }

285    /// Returns the length of a string.
286    ///
287    /// # Safety
288    ///
289    /// The caller must ensure the string is null-terminated and valid.
290    pub(crate) unsafe fn strlen(
291        str: *const libc::c_char,
292    ) -> libc::c_ulong {
293        let mut end = str;
294        while *end != 0 {
295            end = end.add(1);
296        }
297        end.offset_from(str) as libc::c_ulong
298    }
299
300    /// Compares up to `count` characters of two strings.
301    ///
302    /// # Safety
303    ///
304    /// The caller must ensure the strings are null-terminated and valid.
305    pub(crate) unsafe fn strncmp(
306        lhs: *const libc::c_char,
307        rhs: *const libc::c_char,
308        mut count: libc::c_ulong,
309    ) -> libc::c_int {
310        if lhs.is_null() || rhs.is_null() {
311            return die();
312        }
313        let mut lhs = lhs.cast::<u8>();
314        let mut rhs = rhs.cast::<u8>();
315        while count > 0 && *lhs != 0 && *lhs == *rhs {
316            lhs = lhs.add(1);
317            rhs = rhs.add(1);
318            count -= 1;
319        }
320        if count == 0 {
321            0
322        } else {
323            (*lhs).cmp(&*rhs) as libc::c_int
324        }
325    }

327    /// Macro for asserting conditions.
328    ///
329    /// This macro is used internally to assert conditions and handle failures.
330    macro_rules! __assert {
331        (false $(,)?) => {
332            $crate::externs::__assert_fail(
333                stringify!(false),
334                file!(),
335                line!(),
336            )
337        };
338        ($assertion:expr $(,)?) => {
339            if !$assertion {
340                $crate::externs::__assert_fail(
341                    stringify!($assertion),
342                    file!(),
343                    line!(),
344                );
345            }
346        };
347    }
348
349    /// Internal function for handling assertion failures.
350    ///
351    /// # Safety
352    ///
353    /// This function is called when an assertion fails, and it triggers a panic.
354    pub(crate) unsafe fn __assert_fail(
355        __assertion: &'static str,
356        __file: &'static str,
357        __line: u32,
358    ) -> ! {
359        struct Abort;
360        impl Drop for Abort {
361            fn drop(&mut self) {
362                panic!("arithmetic overflow");
363            }
364        }
365        let _abort_on_panic = Abort;
366        panic!(
367            "{}:{}: Assertion `{}` failed.",
368            __file, __line, __assertion
369        );
370    }
371}

373/// Module for formatting utilities.
374///
375/// This module provides utilities for formatting data,
376/// particularly for writing formatted strings to pointers.
377mod fmt {
378    use crate::yaml::yaml_char_t;
379    use core::fmt::{self, Write};
380    use core::ptr;
381
382    /// Struct for writing formatted data to a pointer.
383    pub(crate) struct WriteToPtr {
384        ptr: *mut yaml_char_t,
385    }
386
387    impl WriteToPtr {
388        /// Creates a new `WriteToPtr`.
389        ///
390        /// # Safety
391        ///
392        /// This function is unsafe because it directly manipulates raw pointers.
393        pub(crate) unsafe fn new(ptr: *mut yaml_char_t) -> Self {
394            WriteToPtr { ptr }
395        }
396
397        /// Writes formatted data to the pointer.
398        pub(crate) fn write_fmt(&mut self, args: fmt::Arguments<'_>) {
399            let _ = Write::write_fmt(self, args);
400        }
401    }
402
403    impl Write for WriteToPtr {
404        fn write_str(&mut self, s: &str) -> fmt::Result {
405            unsafe {
406                ptr::copy_nonoverlapping(s.as_ptr(), self.ptr, s.len());
407                self.ptr = self.ptr.add(s.len());
408            }
409            Ok(())
410        }
411    }
412}

414/// Trait extension for pointers.
415///
416/// This trait provides methods for working with pointers,
417/// particularly for calculating the offset between two pointers.
418trait PointerExt: Sized {
419    fn c_offset_from(self, origin: Self) -> isize;
420}
421
422impl<T> PointerExt for *const T {
423    fn c_offset_from(self, origin: *const T) -> isize {
424        (self as isize - origin as isize) / size_of::<T>() as isize
425    }
426}
427
428impl<T> PointerExt for *mut T {
429    fn c_offset_from(self, origin: *mut T) -> isize {
430        (self as isize - origin as isize) / size_of::<T>() as isize
431    }
432}
