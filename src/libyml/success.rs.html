1use core::ops::Deref;
2
3/// Constant representing a successful operation.
4pub const OK: Success = Success { ok: true };
5
6/// Constant representing a failed operation.
7pub const FAIL: Success = Success { ok: false };
8
9/// Structure representing the success state of an operation.
10#[must_use]
11#[derive(Copy, Clone, Debug, Default, PartialEq)]
12pub struct Success {
13    /// Boolean indicating whether the operation was successful.
14    pub ok: bool,
15}
16
17/// Structure representing the failure state of an operation.
18#[derive(Copy, Clone, Debug, Default, PartialEq)]
19pub struct Failure {
20    /// Boolean indicating whether the operation failed.
21    pub fail: bool,
22}
23
24impl Deref for Success {
25    type Target = Failure;
26
27    /// Returns a reference to the corresponding `Failure` instance based on the success state.
28    fn deref(&self) -> &Self::Target {
29        if self.ok {
30            &Failure { fail: false }
31        } else {
32            &Failure { fail: true }
33        }
34    }
35}
36
37/// Checks if the given `Success` result indicates a successful operation.
38///
39/// # Arguments
40///
41/// * `result` - A `Success` struct representing the result of an operation.
42///
43/// # Returns
44///
45/// * `true` if the operation was successful.
46/// * `false` if the operation failed.
47pub fn is_success(result: Success) -> bool {
48    result.ok
49}
50
51/// Checks if the given `Success` result indicates a failure operation.
52///
53/// # Arguments
54///
55/// * `result` - A `Success` struct representing the result of an operation.
56///
57/// # Returns
58///
59/// * `true` if the operation failed.
60/// * `false` if the operation was successful.
61pub fn is_failure(result: Success) -> bool {
62    !result.ok
63}
