1/// Macro that allows to generate easily a custom Debug implementation
2/// for an enum.
3///
4/// `write!` is used under the hood, allowing custom formatting.
5///
6/// ## Examples
7/// ```rust
8/// use osmia::impl_debug;
9///
10/// enum MyEnum {
11///   A,
12///   B,
13///   SomethingCustom(String),
14///   SomethingCustomComplex(String, u32),
15/// }
16///
17/// impl_debug!(
18///   MyEnum,
19///   (
20///     A <=> "a",
21///     SomethingCustom(x) <=> "SomethingCustom({x:?})",
22///     SomethingCustomComplex(x, y) <=> "SomethingCustomComplex({y}, {x:?})",
23///     B <=> "b", // The enum options order can be changed here
24///   )
25/// );
26///
27/// let tests = vec![
28///   (MyEnum::A, "a"),
29///   (MyEnum::B, "b"),
30///   (
31///       MyEnum::SomethingCustom("test".to_string()),
32///       r#"SomethingCustom("test")"#
33///   ),
34///   (
35///       MyEnum::SomethingCustomComplex("test".to_string(), 1),
36///       r#"SomethingCustomComplex(1, "test")"#
37///   )
38/// ];
39/// for (token, expected) in tests {
40///   println!("Testing: {}", expected);
41///   let token_str = format!("{:?}", token);
42///   assert_eq!(token_str, expected);
43/// }
44/// ```
45#[macro_export]
46macro_rules! impl_debug {
47	(
48		$enum_name:ident,
49		(
50			$(
51				$variant:ident$(( $($var:ident),* ))? <=> $value:expr $(,)?
52			),*
53		)
54	) => {
55		impl std::fmt::Debug for $enum_name {
56			fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
57				match self {
58					$(
59						Self::$variant$(( $($var),* ))? => write!(f, $value),
60					)*
61				}
62			}
63		}
64	};
65}
