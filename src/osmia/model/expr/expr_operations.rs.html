18/// Addition
19impl Add for Expr {
20	type Output = Result<Expr, OsmiaError>;
21
22	/// ```rust
23	/// use osmia::Osmia;
24	///
25	/// let mut osmia = Osmia::default();
26	/// assert_eq!(osmia.run_code("{{ \"Hello \" + \"World\" }}").unwrap(), "Hello World");
27	/// assert_eq!(osmia.run_code("{{ \"Hello \" + 123 }}").unwrap(), "Hello 123");
28	/// assert_eq!(osmia.run_code("{{ true + \" Hello\" }}").unwrap(), "true Hello");
29	/// assert_eq!(osmia.run_code("{{ true + false }}").unwrap(), "true");
30	/// assert_eq!(osmia.run_code("{{ 1 + false }}").unwrap(), "true");
31	/// assert_eq!(osmia.run_code("{{ 1.1 + 2 }}").unwrap(), "3.1");
32	/// assert_eq!(osmia.run_code("{{ 1 + 2.1 }}").unwrap(), "3");
33	/// assert_eq!(osmia.run_code("{{ null + null }}").unwrap(), "null");
34	/// assert_eq!(osmia.run_code("{{ [1, 2] + [3, 4] }}").unwrap(), "[1, 2, 3, 4]");
35	/// assert_eq!(osmia.run_code(r#"{{ {"a": 1, "b": 2} + {"b": 3, "c": 4} }}"#).unwrap(), r#"{"a": 1, "b": 3, "c": 4}"#);
36	/// ```
37	fn add(self, rhs: Self) -> Self::Output {
38		match (&self, &rhs) {
39			(Expr::Str(s1), s2) => Ok(Expr::Str(format!("{}{}", s1, s2.to_string()))),
40			(s1, Expr::Str(s2)) => Ok(Expr::Str(format!("{}{}", s1.to_string(), s2))),
41			(Expr::Bool(b1), b2) => Ok(Expr::Bool(*b1 || b2.to_bool())),
42			(b1, Expr::Bool(b2)) => Ok(Expr::Bool(b1.to_bool() || *b2)),
43			(Expr::Float(n1), n2) => Ok(Expr::Float(n1 + n2.to_float()?)),
44			(Expr::Int(n1), n2) => Ok(Expr::Int(
45				n1.checked_add(n2.to_int()?)
46				.ok_or(format!(
47					"Cannot add {} and {}: It will overflow",
48					n1, n2
49				))?
50			)),
51			(Expr::Null, Expr::Null) => Ok(Expr::Null),
52			(Expr::Array(arr1), Expr::Array(arr2)) => Ok(Expr::Array(arr1 + arr2)),
53			(Expr::Object(obj1), Expr::Object(obj2)) => Ok(Expr::Object((obj1 | obj2)?)),
54			(s, rhs) => Err(format!("Don't know how to add {} and {}", s, rhs))
55		}
56	}
57}

297		match (self, other) {
298			(Expr::Int(n1), Expr::Int(n2)) => n1.partial_cmp(n2),
299			(Expr::Int(n1), Expr::Float(_)) => n1.partial_cmp(&other.to_int().unwrap()),
300			(Expr::Float(n1), Expr::Float(n2)) => n1.partial_cmp(n2),
301			(Expr::Float(n1), Expr::Int(_)) => n1.partial_cmp(&other.to_float().unwrap()),
302			(Expr::Str(s1), s2) => s1.partial_cmp(&s2.to_string()),
303			(Expr::Bool(b1), b2) => b1.partial_cmp(&b2.to_bool()),
304			_ => None
305		}
306	}

309fn cast_int_for_operation(e: Expr, operation: &str) -> Result<i64, OsmiaError> {
310	match e.to_int() {

312		Err(e) => Err(format!(
313			"Cannot execute operation {} on non-integer value: {}",
314			operation, e
315		))
316	}

319/// BitAnd
320impl BitAnd for Expr {
321	type Output = Result<Expr, OsmiaError>;
322
323	/// ```rust
324	/// use osmia::Osmia;
325	///
326	/// let mut osmia = Osmia::default();
327	/// assert_eq!(osmia.run_code("{{ 1 & 2 }}").unwrap(), "0");
328	/// assert_eq!(osmia.run_code("{{ 1.2 & 2.1 }}").unwrap(), "0");
329	/// assert_eq!(osmia.run_code("{{ 1.2 & 2 }}").unwrap(), "0");
330	/// assert_eq!(osmia.run_code("{{ 1 & 2.1 }}").unwrap(), "0");
331	/// assert_eq!(osmia.run_code("{{ 1.2 & 2.1 }}").unwrap(), "0");
332	/// assert_eq!(osmia.run_code("{{ 1 & 2 }}").unwrap(), "0");
333	/// assert_eq!(osmia.run_code("{{ true & false }}").unwrap(), "false");
334	/// assert_eq!(osmia.run_code("{{ true & true }}").unwrap(), "true");
335	/// assert_eq!(osmia.run_code("{{ false & true }}").unwrap(), "false");
336	/// assert_eq!(osmia.run_code("{{ false & false }}").unwrap(), "false");
337	/// assert_eq!(osmia.run_code(r#"{{ {"a": 1, "b": 2} & {"b": 3, "c": 4} }}"#).unwrap(), r#"{"b": 2}"#);
338	/// ```
339	fn bitand(self, rhs: Expr) -> Self::Output {
340		match (&self, &rhs) {
341			(Expr::Bool(b1), Expr::Bool(b2)) => return Ok(Expr::Bool(*b1 & *b2)),
342			(Expr::Object(o1), Expr::Object(o2)) => return Ok(Expr::Object((o1 & o2)?)),
343			_ => (),
344		};
345		const OPERATION: &str = "bitand (&)";
346		Ok(Expr::Int(
347			cast_int_for_operation(self, OPERATION)? &
348			cast_int_for_operation(rhs, OPERATION)?
349		))
350	}
351}

353impl BitAnd for &Object {
354	type Output = Result<Object, OsmiaError>;
355
356	fn bitand(self, rhs: Self) -> Self::Output {
357		let mut common_entries = Vec::new();
358		for entry in self.entries() {
359			if rhs.contains_key(&entry.0) {
360				common_entries.push(entry);
361			}
362		}
363		Ok(match self {
364			Object::Code(_) => Object::new_code(common_entries),
365			Object::Hash(_) => Object::new_hash(common_entries)?,
366		})
367	}
368}

453	fn bitxor(self, rhs: Self) -> Self::Output {
454		let mut entries = Vec::new();
455		for entry in self.entries() {
456			if !rhs.contains_key(&entry.0) {
457				entries.push(entry);
458			}
459		}
460		for entry in rhs.entries() {
461			if !self.contains_key(&entry.0) {
462				entries.push(entry);
463			}
464		}
465		Ok(Object::new_code(entries))
466	}
467}

469/// Shl
470impl Shl for Expr {
471	type Output = Result<Expr, OsmiaError>;
472
473	/// ```rust
474	/// use osmia::Osmia;
475	///
476	/// let mut osmia = Osmia::default();
477	/// assert_eq!(osmia.run_code("{{ 7 << 4 }}").unwrap(), "112");
478	/// assert_eq!(osmia.run_code("{{ 1.2 << 2.1 }}").unwrap(), "4");
479	/// assert_eq!(osmia.run_code("{{ 1.2 << 2 }}").unwrap(), "4");
480	/// assert_eq!(osmia.run_code("{{ 1 << 2.1 }}").unwrap(), "4");
481	/// assert_eq!(osmia.run_code("{{ 1.2 << 2.1 }}").unwrap(), "4");
482	/// assert_eq!(osmia.run_code("{{ 1 << 2 }}").unwrap(), "4");
483	/// assert_eq!(osmia.run_code("{{ 1 << 7 }}").unwrap(), "128");
484	/// assert_eq!(osmia.run_code("{{ 1 << 63 }}").unwrap(), "-9223372036854775808");
485	/// assert_eq!(osmia.run_code(r#"{{ "foo" << 1 }}"#).unwrap(), "oo");
486	/// assert_eq!(osmia.run_code(r#"{{ "foo" << 2 }}"#).unwrap(), "o");
487	/// assert_eq!(osmia.run_code(r#"{{ "foo" << 3 }}"#).unwrap(), "");
488	/// assert_eq!(osmia.run_code(r#"{{ "foo" << 10 }}"#).unwrap(), "");
489	/// ```
490	fn shl(self, rhs: Expr) -> Self::Output {
491		match (&self, &rhs) {
492			(Expr::Str(s), Expr::Int(i)) => {
493				if *i < 0 {
494					return Err(format!("Invalid shift amount: {:?}", rhs));
495				}
496				let i: usize = *i as usize;
497				let s: &str = match i < s.len() {
498					true => &s[i..],
499					false => "",
500				};
501				return Ok(Expr::new_str(s));
502			},
503			_ => (),
504		}
505		const OPERATION: &str = "shl (<<)";
506		Ok(Expr::Int(
507			cast_int_for_operation(self, OPERATION)? <<
508			cast_int_for_operation(rhs, OPERATION)?
509		))
510	}
511}
