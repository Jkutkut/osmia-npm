34impl Interpreter<ParserCode, OsmiaOutput, OsmiaError> for OsmiaInterpreter<'_> {
35	fn interpret(&self, code: ParserCode) -> Result<OsmiaOutput, OsmiaError> {
36		match (&code).accept(self)? {
37			(ExitStatus::Okay, r) => Ok(r.to_string()),
38			(ExitStatus::Break, _) | (ExitStatus::Continue, _) => Err("Cannot break or continue out of the program".into()),
39			(ExitStatus::Return, r) => Ok(r.to_string()),
40		}
41	}
42}

54type StmtOutput = (ExitStatus, OsmiaResult);
55type StmtResult = Result<StmtOutput, OsmiaError>;
56type ExprResult = Result<Expr, OsmiaError>;
57
58impl Visitor<StmtResult, ExprResult> for OsmiaInterpreter<'_> {
59	fn visit_stmt(&self, stmt: &Stmt) -> StmtResult {
60		match stmt {
61			Stmt::Raw(s) => Ok((ExitStatus::Okay, OsmiaResult::OsmiaOutput(s.into()))),
62			Stmt::NonPrintable(_) => Ok((ExitStatus::Okay, OsmiaResult::None)),
63			Stmt::Block(b) => self.visit_block(b),
64			// TODO keep track of line
65			Stmt::NewLine => Ok((ExitStatus::Okay, OsmiaResult::OsmiaOutput("\n".into()))),
66			Stmt::NewLineNonPrintable => Ok((ExitStatus::Okay, OsmiaResult::None)),
67			Stmt::Expr(e) => Ok((ExitStatus::Okay, OsmiaResult::Expr(e.accept(self)?))),
68			Stmt::Print(s) => self.visit_print(s),
69			Stmt::Comment(_) => Ok((ExitStatus::Okay, OsmiaResult::None)),
70			Stmt::Assign(a) => self.visit_assign(a),
71			Stmt::If(i) => self.visit_if(i),
72			Stmt::While(w) => self.visit_while(w),
73			Stmt::For(f) => self.visit_for(f),
74			Stmt::Break => Ok((ExitStatus::Break, OsmiaResult::None)),
75			Stmt::Continue => Ok((ExitStatus::Continue, OsmiaResult::None)),
76			Stmt::Return(r) => self.visit_return(r),
77			Stmt::Function(f) => self.visit_function(f),
78		}
79	}
80
81	fn visit_expr(&self, expr: &Expr) -> ExprResult {
82		Ok(match expr {
83			Expr::Float(_) | Expr::Int(_) | Expr::Str(_) | Expr::Bool(_) | Expr::Null => expr.clone(),
84			Expr::Binary(b) => self.visit_binary(b)?,
85			Expr::Grouping(g) => self.visit_grouping(g)?,
86			Expr::Unary(u) => self.visit_unary(u)?,
87			Expr::Array(arr) => self.visit_array(arr)?,
88			Expr::Object(obj) => self.visit_object(obj)?,
89			Expr::Lambda(l) => self.visit_lambda(l)?,
90			Expr::Call(c) => self.visit_call(c)?,
91			Expr::MethodCall(m) => self.visit_method_call(m)?,
92			Expr::Variable(v) => self.get_variable(v)?,
93			Expr::Callable(_) => expr.clone(),
94		})
95	}
96}

98impl OsmiaInterpreter<'_> {
99	fn visit_block(&self, block: &Block) -> StmtResult {
100		let mut state = ExitStatus::Okay;
101		let mut content = String::new();
102		for s in block.stmts() {
103			let (status, r) = s.accept(self)?;
104			content += r.to_string().as_str();
105			state = status;
106			match &state {
107				ExitStatus::Okay => (),
108				ExitStatus::Break | ExitStatus::Continue => break,
109				ExitStatus::Return => match r {
110					OsmiaResult::None => break,
111					_ => return Ok((state, r)),
112				}
113			}
114		}
115		Ok((state, content.into()))
116	}
117
118	fn visit_if(&self, if_stmt: &If) -> StmtResult {
119		if let Some(content) = self.visit_conditional(if_stmt.conditional())? {
120			return Ok(content);
121		}
122		if let Some(else_ifs) = if_stmt.elseifs() {
123			for e in else_ifs {
124				if let Some(content) = self.visit_conditional(e)? {
125					return Ok(content);
126				}
127			}
128		}
129		if let Some(e) = if_stmt.else_block() {
130			let e: &Stmt = &*e;
131			return Ok(e.accept(self)?);
132		}
133		Ok((ExitStatus::Okay, OsmiaResult::None))
134	}
135
136	fn visit_while(&self, while_stmt: &While) -> StmtResult {
137		let mut content = String::new();
138		while let Some((status, r)) = self.visit_conditional(while_stmt)? {
139			content.push_str(r.to_string().as_str());
140			match status {
141				ExitStatus::Okay | ExitStatus::Continue => (),
142				ExitStatus::Break => break,
143				ExitStatus::Return => match r {
144					OsmiaResult::None => break,
145					_ => return Ok((status, r)),
146				}
147			}
148		}
149		Ok((ExitStatus::Okay, content.into()))
150	}
151
152	fn visit_conditional(&self, conditional: &ConditionalStmt) -> Result<Option<StmtOutput>, OsmiaError> {
153		match conditional.condition().accept(self)?.to_bool() {
154			false => Ok(None),
155			true => Ok(Some(conditional.body().accept(self)?)),
156		}
157	}
158
159	fn visit_for(&self, for_stmt: &For) -> StmtResult {
160		let var = Self::var_arr_to_ctx_variable(self.visit_variable(for_stmt.variable())?.vec())?;
161		let iterable = self.visit_iterable(for_stmt.iterable())?;
162		let body = for_stmt.body();
163		let mut content = String::new();
164		for e in iterable {
165			self.set_variable(&var, (&e).try_into()?)?;
166			let (status, r) = body.accept(self)?;
167			content += r.to_string().as_str();
168			match status {
169				ExitStatus::Okay | ExitStatus::Continue => (),
170				ExitStatus::Break => break,
171				ExitStatus::Return => match r {
172					OsmiaResult::None => break,
173					_ => return Ok((status, r)),
174				}
175			}
176		}
177		Ok((ExitStatus::Okay, content.into()))
178	}
179
180	fn visit_print(&self, s: &Print) -> StmtResult {
181		println!("{}", self.visit_expr(s.expr())?);
182		Ok((ExitStatus::Okay, OsmiaResult::None))
183	}
184
185	fn visit_assign(&self, assign: &Assign) -> StmtResult {
186		let var = Self::var_arr_to_ctx_variable(self.visit_variable(assign.variable())?.vec())?;
187		let value: Expr = assign.value().accept(self)?;
188		let value = (&value).try_into()?;
189		self.set_variable(&var, value)?;
190		Ok((ExitStatus::Okay, OsmiaResult::None))
191	}
192
193	fn visit_return(&self, r: &Return) -> StmtResult {
194		let value = match r.expr() {
195			Some(expr) => expr.accept(self)?.into(),
196			None => OsmiaResult::None,
197		};
198		Ok((ExitStatus::Return, value))
199	}
200
201	fn visit_function(&self, ft: &Function) -> StmtResult {
202		let name = ft.name().clone();
203		let params = self.visit_function_params(ft.params())?;
204		let ft = Function::new(ft.name().clone(), params, ft.body().clone());
205		let callable = Callable::Function(FunctionCallable::new(ft));
206		self.set_variable(
207			&vec![name],
208			(&Expr::Callable(callable)).try_into()?,
209		)?;
210		Ok((ExitStatus::Okay, OsmiaResult::None))
211	}
212}

214impl OsmiaInterpreter<'_> {
215	fn visit_binary(&self, binary: &Binary) -> ExprResult {
216		let left = binary.left().accept(self)?;
217		let right = binary.right().accept(self)?;
218		match binary.operator() {
219			BinaryOp::Plus => left + right,
220			BinaryOp::Minus => left - right,
221			BinaryOp::Mult => left * right,
222			BinaryOp::Div => left / right,
223			BinaryOp::Mod => left % right,
224			BinaryOp::Equal => Ok(Expr::Bool(left == right)),
225			BinaryOp::NotEqual => Ok(Expr::Bool(left != right)),
226			BinaryOp::Greater => Ok(Expr::Bool(left > right)),
227			BinaryOp::GreaterEqual => Ok(Expr::Bool(left >= right)),
228			BinaryOp::Less => Ok(Expr::Bool(left < right)),
229			BinaryOp::LessEqual => Ok(Expr::Bool(left <= right)),
230			BinaryOp::BitAnd => left & right,
231			BinaryOp::BitOr => left | right,
232			BinaryOp::BitXor => left ^ right,
233			BinaryOp::BitShiftLeft => left << right,
234			BinaryOp::BitShiftRight => left >> right,
235			BinaryOp::And => Ok(Expr::Bool(left.to_bool() && right.to_bool())),
236			BinaryOp::Or => Ok(Expr::Bool(left.to_bool() || right.to_bool())),
237		}
238	}
239
240	fn visit_grouping(&self, grouping: &Grouping) -> ExprResult {
241		grouping.expr().accept(self)
242	}
243
244	fn visit_unary(&self, unary: &Unary) -> ExprResult {
245		match unary.operator() {
246			UnaryOp::Plus => Ok(unary.expr().accept(self)?.affirm()?),
247			UnaryOp::Minus => Ok((-unary.expr().accept(self)?)?),
248			UnaryOp::Not => Ok(!unary.expr().accept(self)?),
249		}
250	}
251
252	fn visit_array(&self, arr: &Array) -> ExprResult {
253		let mut new_arr = Vec::new();
254		for e in arr.iter() {
255			new_arr.push(e.accept(self)?);
256		}
257		Ok(Expr::Array(new_arr.into()))
258	}
259
260	fn visit_object(&self, obj: &Object) -> ExprResult {
261		match obj {
262			Object::Code(_) => {
263				let items: Vec<(Expr, Expr)> = obj.into();
264				let mut new_obj = Vec::new();
265				for (e, v) in items {
266					new_obj.push((self.visit_expr(&e)?, self.visit_expr(&v)?));
267				}
268				Ok(Expr::Object(Object::new_hash(new_obj)?))
269			},
270			Object::Hash(h) => Ok(self.visit_object(&Object::new_code(h.entries()))?),
271		}
272	}
273
274	fn visit_lambda(&self, lambda: &Lambda) -> ExprResult {
275		let params = self.visit_function_params(lambda.params())?;
276		let lambda = Lambda::new(params, lambda.body().clone());
277		let callable = Callable::Lambda(LambdaCallable::new(lambda));
278		Ok(Expr::Callable(callable))
279	}
280
281	fn visit_function_params(&self, params: &Vec<FunctionParam>) -> Result<Vec<FunctionParam>, OsmiaError> {
282		let mut new_params = Vec::new();
283		for p in params {
284			match p {
285				FunctionParam::Param(p, default) => {
286					let default = match default {
287						None => None,
288						Some(d) => Some(self.visit_expr(d)?),
289					};
290					new_params.push(FunctionParam::Param(p.clone(), default));
291				},
292				FunctionParam::Spread(_) => new_params.push(p.clone()),
293			}
294		}
295		Ok(new_params)
296	}
297
298	fn visit_call(&self, call: &Call) -> ExprResult {
299		match call.callee().accept(self)? {
300			Expr::Callable(c) => self.visit_expr(&self.make_call(&c, call.args())?),
301			e => Err(format!("Expression {} is not callable", e)),
302		}
303	}
304
305	fn visit_method_call(&self, m: &MethodCall) -> ExprResult {
306		let obj = m.obj.as_ref().accept(self)?;
307		let var_type = MethodExpression::try_from(&obj).unwrap_or_else(|_| unreachable!());
308		let var = match m.call.callee() {
309			Expr::Variable(v) => v.vec().get(0).unwrap(),
310			_ => unreachable!()
311		};
312		let call_path = vec![
313			JsonTreeKeyExpr::JsonTreeKey(JsonTreeKey::Key(METHOD_CTX_LOCATION.into())),
314			JsonTreeKeyExpr::JsonTreeKey(JsonTreeKey::Key((&var_type).into())),
315			var.clone()
316		];
317		let mut args = Vec::with_capacity(m.call.args().len() + 1);
318		args.push(obj.clone());
319		args.extend_from_slice(m.call.args());
320		let call_expr: Expr = Variable::from_vec(call_path).into();
321		let call = Call::new(call_expr, args);
322		self.visit_call(&call)
323	}
324
325	fn make_call(&self, call: &Callable, args: &Vec<Expr>) -> ExprResult {
326		let args = self.setup_callable_args(args, call)?;
327		let expr: Expr = match call {
328			Callable::Builtin(_) | Callable::Lambda(_) => call.call(self, &args)?,
329			Callable::Function(_) => {
330				self.ctx.borrow_mut().begin_scope();
331				let ft_body = call.call_stmt(self, &args)?;
332				let (status, r) = self.visit_stmt(&ft_body)?;
333				let result = match status {
334					ExitStatus::Continue | ExitStatus::Break => return Err(format!(
335						"Cannot break or continue out of a function"
336					)),
337					ExitStatus::Okay | ExitStatus::Return => match r {
338						OsmiaResult::None => Expr::Null,
339						OsmiaResult::Expr(e) => e,
340						OsmiaResult::OsmiaOutput(s) => Expr::Str(s),
341					}
342				};
343				self.ctx.borrow_mut().end_scope();
344				result
345			}
346		};
347		Ok(expr)
348	}
349
350	fn setup_callable_args(&self, args: &Vec<Expr>, call: &Callable) -> Result<Vec<Expr>, OsmiaError> {
351		let call_arity = call.arity().unwrap_or(args.len());
352		let mut arguments: Vec<Expr> = Vec::with_capacity(call_arity);
353		match call {
354			Callable::Builtin(b) => match b.params() {
355				Some(params) => self.setup_callable_args_with_params(arguments, args, params),
356				None => {
357					let mut i = 0;
358					while i < call_arity && i < args.len() {
359						arguments.push(self.visit_expr(&args[i])?);
360						i += 1;
361					}
362					Ok(arguments)
363				},
364			},
365			Callable::Lambda(l) => self.setup_callable_args_with_params(
366				arguments, args, l.params()
367			),
368			Callable::Function(f) => self.setup_callable_args_with_params(
369				arguments, args, f.params()
370			),
371		}
372	}
373
374	fn setup_callable_args_with_params(
375		&self,
376		mut arguments: Vec<Expr>,
377		args: &Vec<Expr>,
378		ft_params: &Vec<FunctionParam>
379	) -> Result<Vec<Expr>, OsmiaError> {
380		let mut i = 0;
381		loop {
382			match (args.get(i), ft_params.get(i)) {
383				(_, None) => break,
384				(arg, Some(p)) => match (arg, p) {
385					(_, FunctionParam::Spread(_)) => {
386						arguments.push(Expr::Array(args[i..].to_vec().into()));
387						break;
388					},
389					(Some(arg), _) => arguments.push(self.visit_expr(arg)?),
390					(None, _) => match p {
391						FunctionParam::Param(p, None) => return Err(format!(
392							"Missing argument: {}", p
393						)),
394						FunctionParam::Param(_, Some(d)) => arguments.push(self.visit_expr(d)?),
395						FunctionParam::Spread(_) => unreachable!(),
396					}
397				}
398			}
399			i += 1;
400		}
401		Ok(arguments)
402	}
403
404	fn visit_iterable(&self, iterable: &Expr) -> Result<Vec<Expr>, OsmiaError> {
405		let iterable: Expr = match iterable {
406			Expr::Array(arr) => self.visit_array(arr)?,
407			Expr::Object(obj) => self.visit_object(obj)?,
408			Expr::Variable(v) => match self.get_variable(v)? {
409				Expr::Array(a) => Expr::Array(a),
410				Expr::Object(o) => Expr::Object(o),
411				_ => return Err(format!("Variable {:?} is not iterable", v)),
412			},
413			_ => return Err(format!("Cannot iterate over: {:?}", iterable)),
414		};
415		Ok(match iterable {
416			Expr::Array(arr) => (&arr).into(),
417			Expr::Object(obj) => obj.entries().iter().map(|(k, v)| {
418				Ok(Expr::Object(Object::new_hash(vec![
419					(Expr::Str("key".into()), k.clone()),
420					(Expr::Str("value".into()), v.clone()),
421				])?))
422			}).collect::<Result<Vec<Expr>, OsmiaError>>()?,
423			_ => unreachable!(),
424		})
425	}
426
427	fn visit_variable(&self, variable: &Variable) -> Result<Variable, OsmiaError> {
428		let mut arr: Vec<JsonTreeKeyExpr> = Vec::with_capacity(variable.vec().len());
429		for v in variable.vec() {
430			match v {
431				JsonTreeKeyExpr::Expr(e) => arr.push(self.visit_expr(e)?.into()),
432				k => arr.push(k.clone()),
433			}
434		}
435		Ok(Variable::from_vec(arr))
436	}
437
438	fn var_arr_to_ctx_variable(
439		var: &Vec<JsonTreeKeyExpr>
440	) -> Result<Vec<JsonTreeKey<String>>, OsmiaError> {
441		let mut variable_keys: Vec<JsonTreeKey<String>> = Vec::with_capacity(var.len());
442		for v in var {
443			match v {
444				JsonTreeKeyExpr::JsonTreeKey(k) => variable_keys.push(k.clone()),
445				JsonTreeKeyExpr::Expr(e) => {
446					let key: JsonTreeKey<String> = match e {
447						Expr::Str(s) => JsonTreeKey::Key(s.into()),
448						Expr::Int(i) => {
449							if *i < 0 {
450								return Err(format!("Invalid variable index: {:?}", e));
451							}
452							JsonTreeKey::Index(*i as usize)
453						},
454						_ => return Err(format!("Invalid variable key: {:?}", e)),
455					};
456					variable_keys.push(key);
457				},
458			}
459		}
460		Ok(variable_keys)
461	}
462}

464impl OsmiaInterpreter<'_> {
465	fn get_variable<'a>(&self, variable: &Variable) -> ExprResult {
466		let variable = self.visit_variable(variable)?;
467		match variable.vec().get(0) {
468			Some(JsonTreeKeyExpr::JsonTreeKey(_)) => self.get_variable_from_ctx(&self.ctx.borrow(), &variable.vec()),
469			Some(JsonTreeKeyExpr::Expr(e)) => {
470				let context = Ctx::from(e.try_into()?);
471				let keys: Vec<JsonTreeKeyExpr> = variable.vec().iter().skip(1).map(|k| k.clone()).collect();
472				self.get_variable_from_ctx(&context, &keys)
473			},
474			None => unreachable!()
475		}
476	}
477
478	fn get_variable_from_ctx(&self, ctx: &Ctx, variable: &Vec<JsonTreeKeyExpr>) -> ExprResult {
479		let keys = Self::var_arr_to_ctx_variable(&variable)?;
480		Ok(ctx.get(&keys)?.try_into()?)
481	}
482
483	fn set_variable<'a>(
484		&self,
485		var: &Vec<JsonTreeKey<String>>,
486		value: JsonTree<String, CtxValue>
487	) -> Result<(), OsmiaError> {
488		self.ctx.borrow_mut().set(var, value)
489	}
