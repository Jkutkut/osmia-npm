11/// Represents a YAML loader.
12#[derive(Debug)]
13pub struct Loader<'input> {
14    /// The YAML parser used to parse the input.
15    ///
16    /// The `Parser` type is defined in the `libyml` module and represents
17    /// a low-level YAML parser.
18    ///
19    /// The `'input` lifetime parameter indicates the lifetime of the input data
20    /// being parsed. It ensures that the `Loader` does not outlive the input data.
21    pub parser: Option<Parser<'input>>,
22
23    /// The count of documents parsed by the loader.
24    ///
25    /// This field keeps track of the number of YAML documents encountered during parsing.
26    pub parsed_document_count: usize,
27}
28
29/// Represents a YAML document.
30#[derive(Debug)]
31pub struct Document<'input> {
32    /// The parsed events of the document.
33    ///
34    /// This field contains a vector of `(Event<'input>, Mark)` tuples, where:
35    /// - `Event<'input>` represents a parsed YAML event, such as a scalar, sequence, or mapping.
36    ///   The `'input` lifetime parameter indicates the lifetime of the input data associated
37    ///   with the event.
38    /// - `Mark` represents the position in the input where the event was encountered.
39    pub events: Vec<(Event<'input>, Mark)>,
40
41    /// Any error encountered during parsing.
42    ///
43    /// This field is an optional `Arc<ErrorImpl>`, where:
44    /// - `Arc` is a reference-counted smart pointer that allows multiple ownership of the error.
45    /// - `ErrorImpl` is the underlying error type that holds the details of the parsing error.
46    ///
47    /// If an error occurs during parsing, this field will contain `Some(error)`. Otherwise, it
48    /// will be `None`.
49    pub error: Option<Arc<ErrorImpl>>,
50
51    /// Map from alias id to index in events.
52    ///
53    /// This field is a `BTreeMap` that maps alias ids to their corresponding index in the
54    /// `events` vector.
55    ///
56    /// In YAML, an alias is a reference to a previously defined anchor. When an alias is
57    /// encountered during parsing, its id is used to look up the index of the corresponding
58    /// event in the `events` vector.
59    pub anchor_event_map: BTreeMap<usize, usize>,
60
61    /// Map from alias id to name.
62    ///
63    /// This field is a `BTreeMap` that maps alias ids to their corresponding names.
64    pub anchor_names: BTreeMap<usize, String>,
65}
66
67impl<'input> Loader<'input> {
68    /// Constructs a new `Loader` instance from the given progress.
69    ///
70    /// # Arguments
71    ///
72    /// * `progress` - The progress representing the YAML input.
73    ///
74    /// # Errors
75    ///
76    /// Returns an error if there is an issue reading the input.
77    ///
78    /// # Examples
79    ///
80    /// ```
81    /// use serde_yml::loader::Loader;
82    /// use serde_yml::de::Progress;
83    ///
84    /// let input = "---\nkey: value";
85    /// let progress = Progress::Str(input);
86    /// let loader_result = Loader::new(progress);
87    ///
88    /// assert!(loader_result.is_ok());
89    /// ```
90    pub fn new(progress: Progress<'input>) -> Result<Self> {
91        let input = match progress {
92            Progress::Str(s) => Cow::Borrowed(s.as_bytes()),
93            Progress::Slice(bytes) => Cow::Borrowed(bytes),
94            Progress::Read(mut rdr) => {
95                let mut buffer = Vec::new();
96                if let Err(io_error) = rdr.read_to_end(&mut buffer) {
97                    return Err(error::new(ErrorImpl::IoError(
98                        io_error,
99                    )));
100                }
101                Cow::Owned(buffer)
102            }
103            Progress::Iterable(_) | Progress::Document(_) => {
104                unreachable!()
105            }
106            Progress::Fail(err) => return Err(error::shared(err)),
107        };
108
109        Ok(Loader {
110            parser: Some(Parser::new(input)),
111            parsed_document_count: 0,
112        })
113    }
114
115    /// Advances the loader to the next document and returns it.
116    ///
117    /// # Returns
118    ///
119    /// Returns `Some(Document)` if a document is successfully parsed, or `None` if there are no more documents.
120    ///
121    /// # Examples
122    ///
123    /// ```
124    /// use serde_yml::loader::{Loader, Document};
125    /// use serde_yml::de::Progress;
126    ///
127    /// let input = "---\nkey: value";
128    /// let progress = Progress::Str(input);
129    /// let mut loader = Loader::new(progress).unwrap();
130    /// let document = loader.next_document().unwrap();
131    ///
132    /// assert_eq!(document.events.len(), 4);
133    /// ```
134    pub fn next_document(&mut self) -> Option<Document<'input>> {
135        let parser = match &mut self.parser {
136            Some(parser) => parser,
137            None => return None,
138        };
139
140        let first = self.parsed_document_count == 0;
141        self.parsed_document_count += 1;
142
143        let mut anchors = BTreeMap::new();
144        let mut document = Document {
145            events: Vec::new(),
146            error: None,
147            anchor_event_map: BTreeMap::new(),
148            anchor_names: BTreeMap::new(),
149        };
150
151        let anchor_name = |anchor: &Anchor| {
152            format!("{:?}", anchor)
153                .trim_start_matches("\"")
154                .trim_end_matches("\"")
155                .to_owned()
156        };
157
158        loop {
159            let (event, mark) = match parser.parse_next_event() {
160                Ok((event, mark)) => (event, mark),
161                Err(err) => {
162                    document.error = Some(Error::from(err).shared());
163                    return Some(document);
164                }
165            };
166            let event = match event {
167                YamlEvent::StreamStart => continue,
168                YamlEvent::StreamEnd => {
169                    self.parser = None;
170                    return if first {
171                        if document.events.is_empty() {
172                            document.events.push((Event::Void, mark));
173                        }
174                        Some(document)
175                    } else {
176                        None
177                    };
178                }
179                YamlEvent::DocumentStart => continue,
180                YamlEvent::DocumentEnd => return Some(document),
181
182                YamlEvent::Alias(alias) => match anchors.get(&alias) {
183                    Some(id) => Event::Alias(*id),
184                    None => {
185                        document.error = Some(
186                            error::new(ErrorImpl::UnknownAnchor(mark))
187                                .shared(),
188                        );
189                        return Some(document);
190                    }
191                },
192                YamlEvent::Scalar(mut scalar) => {
193                    if let Some(anchor) = scalar.anchor.take() {
194                        let id = anchors.len();
195                        document.anchor_names.insert(id, anchor_name(&anchor));
196                        document
197                            .anchor_event_map
198                            .insert(id, document.events.len());
199                        anchors.insert(anchor, id);
200                    }
201                    Event::Scalar(scalar)
202                }
203                YamlEvent::SequenceStart(mut sequence_start) => {
204                    if let Some(anchor) = sequence_start.anchor.take() {
205                        let id = anchors.len();
206                        document.anchor_names.insert(id, anchor_name(&anchor));
207                        document
208                            .anchor_event_map
209                            .insert(id, document.events.len());
210                        anchors.insert(anchor, id);
211                    }
212                    Event::SequenceStart(sequence_start)
213                }
214                YamlEvent::SequenceEnd => Event::SequenceEnd,
215                YamlEvent::MappingStart(mut mapping_start) => {
216                    if let Some(anchor) = mapping_start.anchor.take() {
217                        let id = anchors.len();
218                        document.anchor_names.insert(id, anchor_name(&anchor));
219                        document
220                            .anchor_event_map
221                            .insert(id, document.events.len());
222                        anchors.insert(anchor, id);
223                    }
224                    Event::MappingStart(mapping_start)
225                }
226                YamlEvent::MappingEnd => Event::MappingEnd,
227            };
228            document.events.push((event, mark));
229        }
230    }
231}
