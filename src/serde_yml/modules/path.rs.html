1use std::fmt::{self, Display};
2
3/// `Path` represents the path to the current value in the input, like `dependencies.serde.typo1`.
4///
5/// The `Path` enum provides a way to represent different types of paths in a YAML-like structure.
6/// It can be used to track the location of values within the input and provide meaningful error messages.
7///
8/// # Variants
9///
10/// - `Root`: Represents the root path.
11/// - `Seq`: Represents a sequence (array) path with a reference to the parent path and an index.
12/// - `Map`: Represents a map (object) path with a reference to the parent path and a key.
13/// - `Alias`: Represents an alias path with a reference to the parent path.
14/// - `Unknown`: Represents an unknown path with a reference to the parent path.
15#[derive(Copy, Clone, Debug, PartialEq)]
16pub enum Path<'a> {
17    /// Represents the root path.
18    Root,
19    /// Represents a sequence (array) path with a reference to the parent path and an index.
20    Seq {
21        /// The parent path.
22        parent: &'a Path<'a>,
23        /// The index within the sequence.
24        index: usize,
25    },
26    /// Represents a map (object) path with a reference to the parent path and a key.
27    Map {
28        /// The parent path.
29        parent: &'a Path<'a>,
30        /// The key within the map.
31        key: &'a str,
32    },
33    /// Represents an alias path with a reference to the parent path.
34    Alias {
35        /// The parent path.
36        parent: &'a Path<'a>,
37    },
38    /// Represents an unknown path with a reference to the parent path.
39    Unknown {
40        /// The parent path.
41        parent: &'a Path<'a>,
42    },
43}
44
45impl Display for Path<'_> {
46    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
47        /// `Parent` is a helper struct used to format the parent path.
48        ///
49        /// It implements the `Display` trait to recursively format the parent path.
50        struct Parent<'a>(&'a Path<'a>);
51
52        impl Display for Parent<'_> {
53            fn fmt(
54                &self,
55                formatter: &mut fmt::Formatter<'_>,
56            ) -> fmt::Result {
57                match self.0 {
58                    Path::Root => Ok(()),
59                    path => write!(formatter, "{}.", path),
60                }
61            }
62        }
63
64        match self {
65            Path::Root => formatter.write_str("."),
66            Path::Seq { parent, index } => {
67                write!(formatter, r"{}\[{}\]", Parent(parent), index)
68            }
69            Path::Map { parent, key } => {
70                write!(formatter, "{}{}", Parent(parent), key)
71            }
72            Path::Alias { parent } => {
73                write!(formatter, "{}", Parent(parent))
74            }
75            Path::Unknown { parent } => {
76                write!(formatter, "{}?", Parent(parent))
77            }
78        }
79    }
80}
