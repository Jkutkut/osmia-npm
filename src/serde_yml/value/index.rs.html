6/// A type that can be used to index into a `serde_yml::Value`. See the `get`
7/// and `get_mut` methods of `Value`.
8///
9/// This trait is sealed and cannot be implemented for types outside of
10/// `serde_yml`.
11pub trait Index: private::Sealed {
12    /// Return None if the key is not already in the sequence or object.
13    #[doc(hidden)]
14    fn index_into<'v>(&self, v: &'v Value) -> Option<&'v Value>;
15
16    /// Return None if the key is not already in the sequence or object.
17    #[doc(hidden)]
18    fn index_into_mut<'v>(
19        &self,
20        v: &'v mut Value,
21    ) -> Option<&'v mut Value>;
22
23    /// Panic if sequence index out of bounds. If key is not already in the object,
24    /// insert it with a value of null. Panic if Value is a type that cannot be
25    /// indexed into, except if Value is null then it can be treated as an empty
26    /// object.
27    #[doc(hidden)]
28    fn index_or_insert<'v>(&self, v: &'v mut Value) -> &'v mut Value;
29}

31impl Index for usize {
32    fn index_into<'v>(&self, v: &'v Value) -> Option<&'v Value> {
33        match v.untag_ref() {
34            Value::Sequence(vec) => vec.get(*self),
35            Value::Mapping(vec) => {
36                vec.get(Value::Number((*self).into()))
37            }
38            _ => None,
39        }
40    }
41
42    fn index_into_mut<'v>(
43        &self,
44        v: &'v mut Value,
45    ) -> Option<&'v mut Value> {
46        match v.untag_mut() {
47            Value::Sequence(vec) => vec.get_mut(*self),
48            Value::Mapping(vec) => {
49                vec.get_mut(Value::Number((*self).into()))
50            }
51            _ => None,
52        }
53    }
54
55    fn index_or_insert<'v>(
56        &self,
57        mut v: &'v mut Value,
58    ) -> &'v mut Value {
59        loop {
60            match v {
61                Value::Sequence(vec) => {
62                    let len = vec.len();
63                    return vec.get_mut(*self).unwrap_or_else(|| {
64                        panic!(
65                            "cannot access index {} of YAML sequence of length {}",
66                            self, len
67                        )
68                    });
69                }
70                Value::Mapping(map) => {
71                    let n = Value::Number((*self).into());
72                    return map.entry(n).or_insert(Value::Null);
73                }
74                Value::Tagged(tagged) => v = &mut tagged.value,
75                _ => panic!(
76                    "cannot access index {} of YAML {}",
77                    self,
78                    Type(v)
79                ),
80            }
81        }
82    }
83}

85fn index_into_mapping<'v, I>(
86    index: &I,
87    v: &'v Value,
88) -> Option<&'v Value>
89where
90    I: ?Sized + mapping::Index,
91{
92    match v.untag_ref() {
93        Value::Mapping(map) => map.get(index),
94        _ => None,
95    }
96}

98fn index_into_mut_mapping<'v, I>(
99    index: &I,
100    v: &'v mut Value,
101) -> Option<&'v mut Value>
102where
103    I: ?Sized + mapping::Index,
104{
105    match v.untag_mut() {
106        Value::Mapping(map) => map.get_mut(index),
107        _ => None,
108    }
109}
110
111fn index_or_insert_mapping<'v, I>(
112    index: &I,
113    mut v: &'v mut Value,
114) -> &'v mut Value
115where
116    I: ?Sized + mapping::Index + ToOwned + Debug,
117    Value: From<I::Owned>,
118{
119    if let Value::Null = *v {
120        *v = Value::Mapping(Mapping::new());
121        return match v {
122            Value::Mapping(map) => {
123                match map.entry(index.to_owned().into()) {
124                    Entry::Vacant(entry) => entry.insert(Value::Null),
125                    Entry::Occupied(_) => unreachable!(),
126                }
127            }
128            _ => unreachable!(),
129        };
130    }
131    loop {
132        match v {
133            Value::Mapping(map) => {
134                return map
135                    .entry(index.to_owned().into())
136                    .or_insert(Value::Null);
137            }
138            Value::Tagged(tagged) => v = &mut tagged.value,
139            _ => panic!(
140                "cannot access key {:?} in YAML {}",
141                index,
142                Type(v)
143            ),
144        }
145    }
146}

222impl fmt::Display for Type<'_> {
223    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
224        match self.0 {
225            Value::Null => formatter.write_str("null"),
226            Value::Bool(_) => formatter.write_str("boolean"),
227            Value::Number(_) => formatter.write_str("number"),
228            Value::String(_) => formatter.write_str("string"),
229            Value::Sequence(_) => formatter.write_str("sequence"),
230            Value::Mapping(_) => formatter.write_str("mapping"),
231            Value::Tagged(_) => unreachable!(),
232        }
233    }
234}

236// The usual semantics of Index is to panic on invalid indexing.
237//
238// That said, the usual semantics are for things like `Vec` and `BTreeMap` which
239// have different use cases than Value. If you are working with a Vec, you know
240// that you are working with a Vec and you can get the len of the Vec and make
241// sure your indices are within bounds. The Value use cases are more
242// loosey-goosey. You got some YAML from an endpoint and you want to pull values
243// out of it. Outside of this Index impl, you already have the option of using
244// `value.as_sequence()` and working with the Vec directly, or matching on
245// `Value::Sequence` and getting the Vec directly. The Index impl means you can
246// skip that and index directly into the thing using a concise syntax. You don't
247// have to check the type, you don't have to check the len, it is all about what
248// you expect the Value to look like.
249//
250// Basically the use cases that would be well served by panicking here are
251// better served by using one of the other approaches: `get` and `get_mut`,
252// `as_sequence`, or match. The value of this impl is that it adds a way of
253// working with Value that is not well served by the existing approaches:
254// concise and careless and sometimes that is exactly what you want.

256/// Index into a `serde_yml::Value` using the syntax `value[0]` or `value["k"]`.
257///
258/// Returns `Value::Null` if the type of `self` does not match the type of
259/// the index, for example if the index is a string and `self` is a sequence
260/// or a number. Also returns `Value::Null` if the given key does not exist
261/// in the map or the given index is not within the bounds of the sequence.
262///
263/// For retrieving deeply nested values, you should have a look at the
264/// `Value::pointer` method.
265///
266/// # Examples
267///
268/// ```
269/// # use serde_yml::Value;
270/// #
271/// # fn main() -> serde_yml::Result<()> {
272/// let data: serde_yml::Value = serde_yml::from_str(r#"{ x: { y: [z, zz] } }"#)?;
273///
274/// assert_eq!(data["x"]["y"], serde_yml::from_str::<Value>(r#"["z", "zz"]"#).unwrap());
275/// assert_eq!(data["x"]["y"][0], serde_yml::from_str::<Value>(r#""z""#).unwrap());
276///
277/// assert_eq!(data["a"], serde_yml::from_str::<Value>(r#"null"#).unwrap()); // returns null for undefined values
278/// assert_eq!(data["a"]["b"], serde_yml::from_str::<Value>(r#"null"#).unwrap()); // does not panic
279/// # Ok(())
280/// # }
281/// ```
282impl<I> ops::Index<I> for Value
283where
284    I: Index,
285{
286    type Output = Value;
287
288    fn index(&self, index: I) -> &Value {
289        static NULL: Value = Value::Null;
290        index.index_into(self).unwrap_or(&NULL)
291    }
292}

294/// Write into a `serde_yml::Value` using the syntax `value[0] = ...` or
295/// `value["k"] = ...`.
296///
297/// If the index is a number, the value must be a sequence of length bigger
298/// than the index. Indexing into a value that is not a sequence or a
299/// sequence that is too small will panic.
300///
301/// If the index is a string, the value must be an object or null which is
302/// treated like an empty object. If the key is not already present in the
303/// object, it will be inserted with a value of null. Indexing into a value
304/// that is neither an object nor null will panic.
305///
306/// # Examples
307///
308/// ```
309/// # fn main() -> serde_yml::Result<()> {
310/// let mut data: serde_yml::Value = serde_yml::from_str(r#"{x: 0}"#)?;
311///
312/// // replace an existing key
313/// data["x"] = serde_yml::from_str(r#"1"#)?;
314///
315/// // insert a new key
316/// data["y"] = serde_yml::from_str(r#"[false, false, false]"#)?;
317///
318/// // replace a value in a sequence
319/// data["y"][0] = serde_yml::from_str(r#"true"#)?;
320///
321/// // inserted a deeply nested key
322/// data["a"]["b"]["c"]["d"] = serde_yml::from_str(r#"true"#)?;
323///
324/// println!("{:?}", data);
325/// # Ok(())
326/// # }
327/// ```
328impl<I> ops::IndexMut<I> for Value
329where
330    I: Index,
331{
332    fn index_mut(&mut self, index: I) -> &mut Value {
333        index.index_or_insert(self)
334    }
